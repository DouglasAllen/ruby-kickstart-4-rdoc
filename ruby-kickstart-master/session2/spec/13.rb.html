 89
 90    blog1 = Blog.new(Date.parse("2010-05-28"), lissa, "Sailor Venus is my favourite")

 92    blog3 = Blog.new(Date.parse("2011-11-11"), lissa, "Sailor Venus is my favourite")
 93    blog4 = Blog.new(Date.parse("2010-05-28"), josh,  "Sailor Venus is my favourite")
 94    blog5 = Blog.new(Date.parse("2010-05-28"), lissa, "Sailor Venus is my least favourite")
 95    expect(blog1).to     eq blog2 # eq uses the == method

 98    expect(blog1).to_not eq blog5

103RSpec.describe 'User' do
104  specify 'A new user has no blogs' do
105    user = User.new 'QTSort'
106    expect(user.blogs).to eq Array.new
107  end
108
109  it 'can add blogs' do
110    user = User.new 'QTSort'
111    user.add_blog(Date.today, "text")
112    expect(user.blogs.size).to       eq 1
113    expect(user.blogs.first.date).to eq Date.today
114    expect(user.blogs.first.text).to eq 'text'
115  end
116
117  it 'can retrieve its added blogs' do
118    user = User.new 'QTSort'
119    blog = user.add_blog(Date.today, "text")
120    expect(user.blogs.first).to eq blog
121  end
122
123  it 'returns its blogs in order of date' do
124    lissa = User.new 'QTSort'
125    blog1 = lissa.add_blog Date.parse("2010-05-28"), "Sailor Venus is my favourite"
126    blog2 = lissa.add_blog Date.parse("2007-01-02"), "Going dancing!"
127    blog3 = lissa.add_blog Date.parse("2006-01-02"), "For the last time, fuck facebook >.<"
128    blog4 = lissa.add_blog Date.parse("2010-01-02"), "Got a new job, cuz I'm pretty much the best ^_^"
129    expect(lissa.blogs.size).to eq 4
130    blog_dates = lissa.blogs.map { |blog| blog.date }
131    expect(blog_dates).to eq [Date.parse('2010-05-28'), Date.parse('2010-01-02'), Date.parse('2007-01-02'), Date.parse('2006-01-02')]
132    expect(lissa.blogs).to eq [blog1, blog4, blog2, blog3]
133  end
