 3RSpec.describe 'BeerSong' do
 4  def stanza(n)
 5    return n.map { |i| stanza i }.reverse.flatten if n.is_a? Range
 6    return [''] if n == 0
 7    return "One bottle of beer on the wall,\nOne bottle of beer,\nTake one down, pass it around,\nZero bottles of beer on the wall.".split("\n") if n == 1
 8    @translation ||= %w(Zero One Two Three Four Five Six Seven Eight Nine Ten Eleven Twelve Thirteen Fourteen Fifteen Sixteen Seventeen Eighteen Nineteen Twenty Twenty-one Twenty-two Twenty-three Twenty-four Twenty-five Twenty-six Twenty-seven Twenty-eight Twenty-nine Thirty Thirty-one Thirty-two Thirty-three Thirty-four Thirty-five Thirty-six Thirty-seven Thirty-eight Thirty-nine Forty Forty-one Forty-two Forty-three Forty-four Forty-five Forty-six Forty-seven Forty-eight Forty-nine Fifty Fifty-one Fifty-two Fifty-three Fifty-four Fifty-five Fifty-six Fifty-seven Fifty-eight Fifty-nine Sixty Sixty-one Sixty-two Sixty-three Sixty-four Sixty-five Sixty-six Sixty-seven Sixty-eight Sixty-nine Seventy Seventy-one Seventy-two Seventy-three Seventy-four Seventy-five Seventy-six Seventy-seven Seventy-eight Seventy-nine Eighty Eighty-one Eighty-two Eighty-three Eighty-four Eighty-five Eighty-six Eighty-seven Eighty-eight Eighty-nine Ninety Ninety-one Ninety-two Ninety-three Ninety-four Ninety-five Ninety-six Ninety-seven Ninety-eight Ninety-nine)
 9    <<-STANZA.gsub(/^\s+/,'').split("\n")
10      #{@translation[n]} bottles of beer on the wall,
11      #{@translation[n]} bottles of beer,
12      Take one down, pass it around,
13      #{@translation[n-1]} bottle#{n == 2 ? '' : 's'} of beer on the wall.
14    STANZA
15  end
16
17  def check_results_for(n)
18    lines = input_output { BeerSong.new(n).print_song }
19              .split("\n")
20              .reject { |line| line.empty? }
21    expect(lines).to eq stanza(1..n)
22  end
23
24  it 'prints nothing when there is no beer' do
25    output = input_output { BeerSong.new(0).print_song }.chomp
26    expect(output).to eq String.new
27  end
28
29  it 'prints the first stanza when there is 1 bottle of beer' do
30    check_results_for 1
31  end
32
33  it 'prints the second stanza when there are 2 bottles of beer' do
34    check_results_for 2
35  end
36
37  it 'prints the third stanza when there are 3 bottles of beer' do
38    check_results_for 3
39  end
40
41  it 'prints nothing when we have have negative bottles of beer' do
42    output = input_output do
43      BeerSong.new(-1).print_song
44      BeerSong.new(-3).print_song
45      BeerSong.new(-192).print_song
46    end
47    expect(output.split "\n").to eq []
48  end
49
50  it 'prints the entire song when there are 99 bottles of beer' do
51    check_results_for 99
52  end
53
54  it 'caps the beer at 99 when there are more than 99 bottles of beer' do
55    result = stanza 1..99
56    expect(input_output { BeerSong.new( 99).print_song }.split("\n")).to eq result
57    expect(input_output { BeerSong.new(100).print_song }.split("\n")).to eq result
58    expect(input_output { BeerSong.new(101).print_song }.split("\n")).to eq result
59  end
60end
