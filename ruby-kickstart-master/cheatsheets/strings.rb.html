 7# Make a new string
 8"Can interpolate #{1+1}"      # => "Can interpolate 2"
 9'Cannot interpolate #{1+1}'   # => "Cannot interpolate \#{1+1}"
10
11string = <<END_OF_DOC.upcase
12  This is a here document, I can interpolate #{1}
13  It will continue until it finds, the line END_OF_DOC by itself.
14  Notice I can continue doing things on the line the here document is defined in.
15  here I called String#upcase on it.
16END_OF_DOC
17
18string  # => "  THIS IS A HERE DOCUMENT, I CAN INTERPOLATE 1\n  IT WILL CONTINUE UNTIL IT FINDS, THE LINE END_OF_DOC BY ITSELF.\n  NOTICE I CAN CONTINUE DOING THINGS ON THE LINE THE HERE DOCUMENT IS DEFINED IN.\n  HERE I CALLED STRING#UPCASE ON IT.\n"

21# Useful methods
22'abc'.length        # => 3
23'abc' == 'abc'      # => true
24'abc' == 'def'      # => false
25'AbCd'.downcase     # => "abcd"
26'AbCd'.upcase       # => "ABCD"
27" abc   ".strip     # => "abc"
28
29# concatenation
30'hello' + 'world'   # => "helloworld"
31'hello' << 'world'  # => "helloworld"
32# wait O.o why two methods for this?
33s = "abc"
34s + "def"       # => "abcdef"
35s               # => "abc"
36s << "def"      # => "abcdef"
37s               # => "abcdef"

39# Access parts of the string (-1 counts backwards from the end)
40"abc"[-1]           # => "c"
41"abc"[0]            # => "a"
42"abc"[1]            # => "b"
43"abc"[2]            # => "c"
44"abc"[3]            # => nil
45"abcdefg"[2..4]     # => "cde"
46"abcdefg"[4..-1]    # => "efg"
47
48# Changing parts of a string
49message = "Josh said"
50message["Josh"] = "dom"
51message             # => "dom said"
52message += ' "Howdy"'
53message             # => "dom said \"Howdy\""
54message.capitalize  # => "Dom said \"howdy\""
55message             # => "dom said \"Howdy\""
56message[-2] = 'e'
57message             # => "dom said \"Howde\""
58message[-2] = 'y!'
59message             # => "dom said \"Howdy!\""

61# Other useful methods (some of these use regular expressions, we'll talk about them more later)
62"abc" * 3                                     # => "abcabcabc"
63" abc \n".chomp                               # => " abc "
64'trite'.gsub('t','b')                         # => "bribe"
65"Their IDs are 11, and 492".scan(/\d+/)       # => ["11", "492"]
66"Jack and Jill ran up the hill".split         # => ["Jack", "and", "Jill", "ran", "up", "the", "hill"]
67"/usr/bin:/opt/local:/Users/josh".split(':')  # => ["/usr/bin", "/opt/local", "/Users/josh"]
68"abcdefg hi".split(//)                        # => ["a", "b", "c", "d", "e", "f", "g", " ", "h", "i"]
