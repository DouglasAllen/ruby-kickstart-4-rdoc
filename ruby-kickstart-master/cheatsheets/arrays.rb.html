 54  # a bang on the end of a method means "watch out", many methods have a "safe" version which will return a new object
 55  # and a "dangerous" version, which will mutate the current object. (this is not it's only use, so check the docs ;)
 56before       # => [3, 6, 3, 0, 8, 235, -3]
 57before.sort!
 58before       # => [-3, 0, 3, 3, 6, 8, 235]
 59
 60  # you can sort based on your own criteria, your block must simply evaluate to -1 , 0 , or 1
 61  # This is the same as Java's compareTo method. We have a method like that already, our spaceship operator
 62  # if this example is too confusing, let me know (or if it's cake, you can brag)
 63after = before.sort do |a,b|
 64  b <=> a   # sort such that larger elements come first
 65end
 66before # => [-3, 0, 3, 3, 6, 8, 235]
 67after  # => [235, 8, 6, 3, 3, 0, -3]

 69  # An object with a name and age
 70class Person
 71  attr_accessor 'name', 'age'
 72  def initialize(name, age)
 73    @name, @age = name, age
 74  end
 75  def inspect
 76    "<#{name}(#{age})>"
 77  end
 78end 
 79
 80  # whatever the block returns will be used to sort the object
 81people = [
 82  Person.new( 'Ernie'    , 20 ),
 83  Person.new( 'Sara'     , 50 ),
 84  Person.new( 'Monique'  , 10 ),
 85  Person.new( 'Mohammed' , 60 ),
 86]
 87people.sort_by { |person| person.age } # => [<Monique(10)>, <Ernie(20)>, <Sara(50)>, <Mohammed(60)>]
