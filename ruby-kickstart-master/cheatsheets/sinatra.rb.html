  9# =====  To Create a Sinatra App  =====
 10# Make a simple text file that ends with .rb
 11# Require rubygems and sinatra
 12# Define your http methods, their routes and responses
 13# Run it like any Ruby file (if shotgun works on your computer, try with that)

 16# =====  HTTP Methods  =====
 17# get , post , put , delete
 18# To define one, simply type which one it is (correlates to a method invocation)
 19# Then pass it the route you wish to match, and the block that defines the code to execute when that route is matched
 20
 21# Now when someone makes a get request to 'yourapp.com/home' it will return 'hello world'
 22get '/home' do
 23  'hello world!'
 24end

 27# =====  Routes  =====
 28# Routes are compiled into regular expressions.
 29# You can place named variables in the route by listing them like symbols
 30# '/hello/:name' will take the name out of the url and make it available in the params hash
 31# Routes are relative to the path your app is mounted on (probably the root of your application, ie localhost)
 32
 33# what do the routes look like when they get compiled?
 34module Sinatra
 35  class Base
 36    # Pure routes
 37    compile '/'                               # => [/^\/$/, []]
 38    compile '/actors/brad-pitt'               # => [/^\/actors\/brad(?:\-|%2D)pitt$/, []]
 39    
 40    # Routes with variables - notice it replaces the variable name with a capture group, 
 41    # And it keeps track of the variable name for assigning the match in the params hash
 42    compile '/actors/:firstname-:lastname'    # => [/^\/actors\/([^\/?#]+)(?:\-|%2D)([^\/?#]+)$/, ["firstname", "lastname"]]
 43    compile '/it/is/:time/in/:location'       # => [/^\/it\/is\/([^\/?#]+)\/in\/([^\/?#]+)$/, ["time", "location"]]
 44    
 45    # The splat will greedily eat up as much as it can, and get placed in the params hash under the name 'splat'
 46    compile '/the-path-is/*'                  # => [/^\/the(?:\-|%2D)path(?:\-|%2D)is\/(.*?)$/, ["splat"]]
 47  end
 48end
 49
 50# If you want, you can even create your own regexes for the routes
 51# Then, instead of naming them after the symbol given in the route, it will be passed into the block
 52get %r{/hello/([\w]+)} do |firstname|
 53  "Hello, #{firstname}!"
 54end

 58# =====  The Params Hash  =====
 59# You can access the params hash by simply calling the params method, which returns the hash.
 60# The hash is populated with the parameters passed to your program
 61# They may come from matching the route you gave
 62# They may be passed at the end of a route
 63# They may be submitted with form data
 64actor_acknowledgment = lambda do 
 65  firstname  =  params[:firstname].capitalize
 66  lastname   =  params[:lastname].capitalize
 67  "You are looking for information about #{firstname} #{lastname}!"
 68end
 69
 70# Each of the following url and method pairs will return "You are looking for information about Brad Pitt"
 71# When you go to them in your browser (the last one is done with curl -- sry Windows guys, didn't know how else to show it)
 72
 73# http://localhost:4567/actors?lastname=pitt&firstname=brad
 74# http://localhost:4567/actors?firstname=brad&lastname=pitt
 75get '/actors' , &actor_acknowledgment
 76
 77# http://localhost:4567/actors/brad-pitt
 78# (probably not a great choice since real people can have hyphens in their name :P)
 79get '/actors/:firstname-:lastname' , &actor_acknowledgment
 80
 81# http://localhost:4567/actors/brad/pitt
 82get '/actors/:firstname/:lastname' , &actor_acknowledgment
 83
 84# curl -d 'firstname=brad&lastname=pitt' http://localhost:4567/actors
 85# Notice in this case, that firstname and lastname will be passed invisibly ;)
 86post '/actors' , &actor_acknowledgment

 89# =====  The Block  =====
 90# When defining how to respond to a request, you submit the http method, the route, and the block
 91# The block is what will be executed when the given http method 
 92# Your params hash will be set and available, and any parameters that were captured will be in there
 93# Whatever your block returns is what will be displayed to the user
 94

 96# -----  Redirect  -----
 97# You can send an HTTP redirect with the redirect method
 98get '/invalid' do
 99  redirect '/valid'
100end
101

103  redirect 'http://www.ruby-lang.org/'

107# -----  Cookie Sessions  -----
108# Sessions are a way that you can remember a user between HTTP requests (remember, HTTP is inherently stateless)
109# The easiest way is to use cookies. Cookies are very small (<4kb) text files that sit on the client's computer
110# Also because they are stored on the client's computer, they can be edited by the client, so be careful what you put in there
111
112# To use cookies for sessions, you need to specify so outside the block (ie at the top of your file)
113set :sessions, true
114
115# Sessions are implemented with a simple hash
116get '/' do
117  session["thing"] ||= 0
118  session["thing"] += 1
119  "Thing is now: #{session["thing"]}"
120end

123# =====  Before Filters  =====
124# Remember challenge 5:2 where my boss was using before filters to set @body_class? Sinatra has something similar. 
125# You can specify a block of code to be run before each request. Instance variables are visible, and it can modify the environment
126
127# for example, with the given get request, if we went to '/rock' in our browser, it would work. But if we went to '/rock/' it would not
128# because it would no longer match the regular expression generated by the path
129# but we could make a before filter that modifies the path to always remove any trailing slashes
130before { request.path_info.gsub! /\/$/ , '' } # !> ambiguous first argument; put parentheses or even spaces
131get( '/rock' ){ 'roll' }

134# =====  Views  =====
135# As your Sinatra applications become more complex, creating Strings in your blocks will become britle, obtrusive
136# You can add views to a data section within the same file (you'll have to look that up on your own if you want to try it)
137# You can also place them into their own files in a subdirectory called 'views'
138# then if you want to call it from your sinatra block, you just call the method named after the templating language,
139# and pass in the name of your template
140
141get '/' do
142  @people = %w(bill barry sally sue jerry quinn)
143  erb :roster # I return a String, so still need last line
144end
145

149#   <% @people.each do |person| %>
150#     <li>
151#       <%= person %>
152#     </li>
153#   <% end %>
154# <ul>

157# =====  Use A Layout  =====
158# Want to give your code a layout that each result is dropped into?
159# Just create a file views/layout.erb
160# Sinatra will know to use erb to render it b/c of your extension
161# Like with blocks, call yield when you want the content to be included
162
163# -----  File: views/layout.erb  -----
164# <!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Transitional//EN" "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd">
165# <html lang='en' xml:lang='en' xmlns='http://www.w3.org/1999/xhtml'>
166#   <head>
167#     <title>My Awesome App!</title>
168#     <meta content='text/html;charset=utf-8' http-equiv='Content-Type' />
169#   </head>
170#   <body>
171#     <%= yield %>
172#   </body>
173# </html>
174
175
176# what if you don't want to use the layout?
177get '/roster' do
178  @roster = %w(amy bill george sally)
179  erb :roster , :layout => false
180end
