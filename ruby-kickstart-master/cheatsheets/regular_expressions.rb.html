  3# =====  Quickreference  =====
  4# .             any character except newline
  5# [ ]           any single character of set
  6# [^ ]          any single character NOT of set
  7# *             0 or more previous regular expression
  8# *?            0 or more previous regular expression (non-greedy)
  9# +             1 or more previous regular expression
 10# +?            1 or more previous regular expression (non-greedy)
 11# ?             0 or 1 previous regular expression
 12# |             alternation
 13# ( )           grouping regular expressions
 14# ^             beginning of a line or string
 15# $             end of a line or string
 16# {m,n}         at least m but most n previous regular expression
 17# {m,n}?        at least m but most n previous regular expression (non-greedy)

 19# \&            whole match
 20# \`            pre-match
 21# \'            post-match

 52# if you have a bunch of / in your regex, you would have to escape them, so that they don't look like the end of the regex
 53# so you can also create a regex with %r followed by a delimiter (just like %w for Arrays of Strings)
 54%r(regex)     # => /regex/

 65# =====  Ways to Use Regex (simple)  =====
 66str = "This is a cheatsheet for regular expressions."
 67
 68# get the index of a match
 69str =~ /h../ # => 1
 70str =~ /q../ # => nil
 71
 72# get the content of a match
 73str[/h../]  # => "his"
 74str[/q../]  # => nil
 75
 76# substitute the first match with sub, and every match with gsub (the g means to apply the regex "globally" to the String -- g is a flag in some other languages)
 77str = "bobbling babbling bubbles"
 78str.sub(  /b.b/ , "--replacement--" )       # => "--replacement--bling babbling bubbles"
 79str.gsub( /b.b/ , "--replacement--" )       # => "--replacement--bling --replacement--bling --replacement--bles"
 80str.gsub( /b.b/ ) { |match| match.upcase }  # => "BOBbling BABbling BUBbles"
 81
 82# find all occurrences of a match
 83str.scan(/b.b/)   # => ["bob", "bab", "bub"]
 84
 85# use the match as a delimiter, and find everything inbetween
 86str.split(/b.b/)  # => ["", "bling ", "bling ", "bles"]
 87
 88# These don't directly work on Symbols, but if you invoke the grep method on an Array
 89# it will convert them to Symbols before checking your regular expression
 90# This will go through each element, and if the regex matches it, that element will be in the return array
 91words = %w(Oh, oh, oh, finding the answer. Oh, oh, oh, oh, oh. Finding that there ain't no answer to find.)
 92words.grep(/e/) # => ["the", "answer.", "there", "answer"]

 96# =====  Flags  =====
 97# You can append a flag to the end of a regex to alter its behaviour
 98# /i         case insensitive
 99# /o         only interpolate #{} blocks once
100# /m         multiline mode - '.' will match newline
101# /x         extended mode - whitespace is ignored
102# /[neus]    encoding: none, EUC, UTF-8, SJIS, respectively
103
104"BOBble bobble".scan(/b.b/)  # => ["bob"]
105"BOBble bobble".scan(/b.b/i) # => ["BOB", "bob"]
106
107str = "bobble\nbubble\nbabble"
108str[/e.b/]  # => nil
109str[/e.b/m] # => "e\nb"

113# =====  Matching Characters  =====
114# Any character by itself will match that character
115"abc"[/a/]  # => "a"
116"abc"[/ab/] # => "ab"
117"abc"[/ac/] # => nil
118"abc"[/A/]  # => nil
119"abc"[/A/i] # => "a"
120
121# A dot will match any character except newline
122"ab\ncd".scan(/./) # => ["a", "b", "c", "d"]
123
124# Brackets are how you define a set of characters that are allowed to match one particular character
125str = "bobble bubble babble"
126str.scan(/b.b/)     # => ["bob", "bub", "bab"]
127str.scan(/b[oa]b/)  # => ["bob", "bab"]
128
129# you can also give ranges when inside of brackets by specifying a start and end character
130# this is the same as /b[abcdefghijklmnopqrst]b/
131str.scan(/b[a-t]b/)                                                 # => ["bob", "bab"]
132# this is the same as /b[abcnop]b/
133str.scan(/b[a-cn-p]b/)                                              # => ["bob", "bab"]
134alphabet = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ"
135alphabet.scan(/[a-cmnoy-z]/)                                        # => ["a", "b", "c", "m", "n", "o", "y", "z"]
136
137
138# To match anything NOT in the set, put a caret at the beginning of the brackets
139str.scan(/b[^o]b/) # => ["bub", "bab"]

153# =====  Quantifiers  =====
154# To continue to match 0 or more previous regular expression, use a spat
155# in this case, we will match digits that occur in series of 0 or more, if they follow the text MX
156# notice that they are greedy, they try to match as much number as they can
157str = "The product tags are plain MX, MX104, MX952, and MX7511."
158str.scan(/MX[1-9]*/) # => ["MX", "MX1", "MX952", "MX7511"]
159
160# the plus does the same thing as the splat, but matches 1 or more insetad of 0 or more
161str.scan(/MX[1-9]+/) # => ["MX1", "MX952", "MX7511"]
162
163# to make them non greedy, so that they match as little as they can instead of as much as they can, put a question mark after them
164str = "Then I said 'Oh no' and he said 'oh yes!', can you believe it?"
165str[/'.*'/]   # => "'Oh no' and he said 'oh yes!'"
166str[/'.*?'/]  # => "'Oh no'"
167
168# Use a question mark to match 0 or 1 time
169str = "The product tags are plain MX, MX104, MX952, and MX7511."
170str.scan(/MX[0-9]?/) # => ["MX", "MX1", "MX9", "MX7"]
171
172# specify a minimum and maximum number of times to look for whatever is to the left by using curly braces
173numbers = "1 12 123 1234 12345 123456 1234567 12345678 123456789 1234567890"
174numbers.scan(/\d{4,8}/) # => ["1234", "12345", "123456", "1234567", "12345678", "12345678", "12345678"]
175# Whoops, things are a little tricky! That isn't what I meant to match at all, it is matching the "1245678" out of things like "1234567890"
176numbers.scan(/\b\d{4,8}\b/) # => ["1234", "12345", "123456", "1234567", "12345678"]

180# =====  Logic, Grouping And Control  =====
181# use a single pipe to express alternation (this one or that one)
182str = "I'm planning a trip to the moon, and then from there to venus, and from there to the center of milky way!"
183str.scan(/moon|venus|milky way/) # => ["moon", "venus", "milky way"]
184
185# you can group things together with parentheses
186str = "I rode a sunbeam from the sunburst on Sunday."
187str.gsub(/sun(beam|burst|day)/i) { |match| match.upcase } # => "I rode a SUNBEAM from the SUNBURST on SUNDAY."
188
189# Parentheses capture whatever is matched inside of them. You can access the capture in Ruby with the variables $1 - $9
190str = "location: France, language: French, name: Aimée"
191str =~ /location: (\w+), language: (\w+), name: (\w+)/
192$1  # => "France"
193$2  # => "French"
194$3  # => "Aimée"
195$4  # => nil
196
197# When you need to refer back to a capture group in a String, you can use \1 through \9
198str = "I rode a sunbeam from the sunburst on Monday."
199str.gsub( /(sun|mon)(beam|burst|day)/i , '(\1)(\2)' ) # => "I rode a (sun)(beam) from the (sun)(burst) on (Mon)(day)."
200# warning, double quoted strings will require you to escape the slash in front of 1 and 2:
201'(\1)(\2)'    # => "(\\1)(\\2)"
202"(\\1)(\\2)"  # => "(\\1)(\\2)"
203
204# If you don't want the group to capture, place ?: at the front of it. Now, the second group is the first capture group.
205str = "I rode a sunbeam from the sunburst on Monday."
206str.gsub( /(?:sun|mon)(beam|burst|day)/i , '(\1)(\2)' ) # => "I rode a (beam)() from the (burst)() on (day)()."
207
208# If you want to conditionally match, if you see something ahead (in 1.9, you can also match behind)
209# then use (?=) and if you want to match if you do not see something ahead, use (?!)
210# NOTE THAT THESE DO NOT TAKE UP ANY WIDTH IN THE MATCH
211str = "This is the firstday of the secondstone of the month."
212str.scan(/(?:first|second)(?:stone|day)/) # => ["firstday", "secondstone"]
213str.scan(/(?:first|second)(?=stone)/) # => ["second"]
214str.scan(/(?:first|second)(?!stone)/) # => ["first"]

218# =====  Zero Width Matchers  =====
219# caret to match the beginning of a line, and dollar sign to match end of a line
220# note that in this example, we don't need multiline because the match itself is not going across multiple lines
221str = "Hello my\ndarling how\nare you?"
222str.scan(/^./)      # => ["H", "d", "a"]
223str.scan(/.$/)      # => ["y", "w", "?"]
224str.scan(/.$.^./)   # => []
225str.scan(/.$.^./m)  # => ["y\nd", "w\na"]
226
227# \A and \Z are just like ^ and $, but they match the beginning and end of the entire string, rather than just the line
228str.scan(/\A./)     # => ["H"]
229str.scan(/.\Z/)     # => ["?"]
230
231# to match the break between a word and nonword boundry, use \b
232str.scan(/\w\b/) # => ["o", "y", "g", "w", "e", "u"]
233str.scan(/\b\w/) # => ["H", "m", "d", "h", "a", "y"]
