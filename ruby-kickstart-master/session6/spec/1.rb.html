  8def unrecoverable_error(message)
  9  message     =  "* #{message} *"
 10  border      =  message.gsub(/./,'*')
 11  blank_line  = ''
 12  puts blank_line, border, message, border, blank_line
 13  exit 1
 14end

 68  # Super displeased with this solution.

 80RSpec.describe "Your application's library" do
 81  specify 'is be in a folder named lib/caesar_cipher.rb' do
 82    assert File.exist?("#{app_root}/lib/caesar_cipher.rb")
 83    require "#{app_root}/lib/caesar_cipher"
 84  end
 85
 86  specify 'Should be in a module named CaesarCipher' do
 87    expect(CaesarCipher).to be_an_instance_of Module
 88  end
 89
 90  describe 'CaesarCipher.encrypt' do
 91    def self.assert_encrypts(encryptions)
 92      encryptions.each do |key, value|
 93        specify("#{key.inspect} encrypts to #{value.inspect}") do
 94          expect(CaesarCipher.encrypt key).to eq value
 95        end
 96      end
 97    end
 98    describe 'with a lowercase message' do
 99      assert_encrypts 'a' => 'd',
100                      'b' => 'e',
101                      'x' => 'a',
102                      'z' => 'c'
103    end
104    describe 'with an uppercase message' do
105      assert_encrypts 'A' => 'D',
106                      'B' => 'E',
107                      'X' => 'A',
108                      'Z' => 'C'
109    end
110    describe 'with a numerical message' do
111      assert_encrypts '0' => '3',
112                      '1' => '4',
113                      '7' => '0',
114                      '9' => '2'
115    end
116    describe 'with non alphanumeric message' do
117      '~!@#$%^&*(){}[];\':",./\\ '.each_char do |char|
118        specify { expect(CaesarCipher.encrypt char.dup).to eq char }
119      end
120    end
121    describe 'with a full message' do
122      specify do
123        expect(CaesarCipher.encrypt "The quick brown fox jumps over the lazy dog.")
124          .to eq "Wkh txlfn eurzq ira mxpsv ryhu wkh odcb grj."
125      end
126    end
127  end
128
129  describe 'CaesarCipher.decrypt' do
130    def self.assert_decrypts(decryptions)
131      decryptions.each do |key, value|
132        specify("#{key.inspect} decrypts to #{value.inspect}") do
133          expect(CaesarCipher.decrypt key).to eq value
134        end
135      end
136    end
137    describe 'with a lowercase message' do
138      assert_decrypts 'd' => 'a',
139                      'e' => 'b',
140                      'a' => 'x',
141                      'c' => 'z'
142    end
143    describe 'with an uppercase message' do
144      assert_decrypts 'D' => 'A',
145                      'E' => 'B',
146                      'A' => 'X',
147                      'C' => 'Z'
148    end
149    describe 'with a numerical message' do
150      assert_decrypts '3' => '0',
151                      '4' => '1',
152                      '0' => '7',
153                      '2' => '9'
154    end
155    describe 'with non alphanumeric message' do
156      '~!@#$%^&*(){}[];\':",./\\ '.each_char do |char|
157        assert_decrypts char.dup => char
158      end
159    end
160    describe 'with a full message' do
161      assert_decrypts "Wkh txlfn eurzq ira mxpsv ryhu wkh odcb grj." =>
162                      "The quick brown fox jumps over the lazy dog."
163    end
164  end
165end
