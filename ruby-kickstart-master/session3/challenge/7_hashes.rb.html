1# This challenge is based off of problem 1
 2# modify it such that it also accepts in the hash, a color (whose values are red("#FF0000"), green("#00FF00"), and blue(#0000FF) )
 3# if the color is set, then it should show up in the style
 4# It should also not be necessary to pass in the hash, if you don't want to specify options
 5#
 6#
 7# EXAMPLE:
 8#
 9# html = HTMLTag.new 'p', 'Hello World', :multiline => false, :color => :red
10# html.to_s     # => <p style="color:#FF0000">Hello World</p>
11#
12# sports = [
13#   HTMLTag.new('li', 'baseball', :multiline => false, :color => :red,   :font => :serif) ,
14#   HTMLTag.new('li', 'soccer',   :multiline => false, :color => :green, :font => :sans_serif) ,
15#   HTMLTag.new('li', 'football', :multiline => false, :color => :blue,  :font => :monospace) ,
16# ]
17#
18# ordered_list = HTMLTag.new 'ol' , sports.join , :multiline => true
19#
20# puts ordered_list  # => (as output)
21#                      <ol >
22#                      <li style='color:#FF0000;font-family:"Times New Roman", "Georgia"'>baseball</li>
23#                      <li style='color:#00FF00;font-family:"Arial", "Verdana"'>soccer</li>
24#                      <li style='color:#0000FF;font-family:"Courier New", "Lucida Console"'>football</li>
25#                      </ol>
26
27class HTMLTag
28  FONTS = {
29    :serif      => '"Times New Roman", "Georgia"',
30    :sans_serif => '"Arial", "Verdana"',
31    :monospace  => '"Courier New", "Lucida Console"'
32  }
33
34  attr_accessor :name, :innerHTML, :options
35
36  # options: :multiline should be true or false
37  def initialize(name, innerHTML, options)
38    @name, @innerHTML, @options = name, innerHTML, options
39  end
40
41  def font
42    font = options[:font]  #  one of :serif, :sans_serif, or :monospace
43    FONTS[font]
44  end
45
46  def style
47    return nil unless options[:font]
48    "style='font-family:#{font}'"
49  end
50
51  def to_s
52    line_end = if options[:multiline] then "\n" else "" end
53    "<#{name} #{style}>#{line_end}"  \
54    "#{innerHTML.chomp}#{line_end}"  \
55    "</#{name}>\n"
56  end
57
58end
