  3# overriding Rake method that displays tasks so that -T will display in the correct order
  4module Rake
  5  class Application
  6    def display_tasks_and_comments
  7      displayable_tasks = tasks.select { |t| t.comment && t.name =~ options.show_task_pattern }.sort do
  8        |t1,t2|
  9        one_is_session = t1.name[/^\d+:(\d+|all)$/]
 10        two_is_session = t2.name[/^\d+:(\d+|all)$/]
 11        if one_is_session && two_is_session
 12          session1 , prb1 = t1.name.split ':'
 13          session2 , prb2 = t2.name.split ':'
 14          if session1 == session2
 15            if prb1 == 'all' then 1 elsif prb2 == 'all' then -1 else prb1.to_i <=> prb2.to_i end
 16          else
 17            session1.to_i <=> session2.to_i
 18          end
 19        elsif one_is_session
 20          -1
 21        elsif two_is_session
 22          1
 23        else
 24          t1.name <=> t2.name
 25        end
 26      end
 27      if options.full_description
 28        displayable_tasks.each do |t|
 29          puts "rake #{t.name_with_args}"
 30          t.full_comment.split("\n").each do |line|
 31            puts "    #{line}"
 32          end
 33          puts
 34        end
 35      else
 36        width = displayable_tasks.collect { |t| t.name_with_args.length }.max || 10
 37        max_column = truncate_output? ? terminal_width - name.size - width - 7 : nil
 38        displayable_tasks.each do |t|
 39          printf "#{name} %-#{width}s  # %s\n",
 40            t.name_with_args, max_column ? truncate(t.comment, max_column) : t.comment
 41        end
 42      end
 43    end
 44  end
 45end

 48# overriding makedirs so that I can pass a path to make them from
 49# pass the path in a hash at the end if desired
 50# ie makedirs 'abc' , 'def' , :path => 'ghi'
 51def makedirs(*dirs)
 52  path = dirs.pop[:path].sub(%r[/$],'')   if dirs.last.is_a? Hash
 53  dirs.map! { |dir| "#{path}/#{dir}" }    if path
 54  super(dirs)
 55end

 58# runs the spec against the challenge
 59# if the environment variable 'solved' has been set,
 60# then it runs them against the solved challenge instead
 61def run_spec(session_num,problem)
 62  sess = "#{File.dirname(__FILE__)}/session#{session_num}"
 63  if ENV['solved']
 64    problem_dir = "#{sess}/solved/#{problem}.rb"
 65  else
 66    problem_dir = Dir["#{sess}/challenge/#{problem}_*.rb"].first
 67  end
 68  sh "ruby -c '#{problem_dir}'"                                                             # check syntax
 69  sh "rspec -cr '#{problem_dir}' -r enumerator '#{sess}/spec/#{problem}.rb' --fail-fast"    # run spec (c for colour, r to require the files, enumerator required for 1.8.6 compatibility)
 70end

 73# returns array of sessions
 74def get_sessions
 75  Dir['*'].grep(/^session\d+$/).map { |folder_name| folder_name[/\d+$/] }
 76end
 77
 78
 79# returns hash where each key is the session to test
 80# each value is a range indicating the problems available for this session
 81def sessions_with_problems
 82  get_sessions.inject Hash.new do |sessions_with_sections, session|
 83    challenges = Dir["session#{session}/challenge/*.rb"]
 84    sessions_with_sections[session] = challenges.map { |challenge| challenge[%r([^/]+?$)].to_i }.sort
 85    sessions_with_sections
 86  end
 87end
 88
 89
 90# create tasks for each challenge
 91sessions_with_problems.each do |session, problems|
 92  namespace session do
 93    problems.each do |problem|
 94      desc "Test session#{session} problem#{problem}"
 95      task(problem) { run_spec session , problem }
 96    end
 97    desc "Test session#{session} all problems"
 98    task :all => problems.to_a
 99  end
100end
101
102task :default => get_sessions.map { |session| "#{session}:all" }
