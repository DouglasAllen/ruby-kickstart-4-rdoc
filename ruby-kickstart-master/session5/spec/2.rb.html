 8class ApplicationController
 9  class << self
10    def before_filter(options=Hash.new,&block)
11      before_filter_options options       # a method we stub to see that it gives what we expect
12      before_filters << block if block
13    end
14    def before_filters
15      @to_invoke ||= Array.new
16    end
17    def invoke_before_filters
18      instance = self.new
19      before_filters.each { |filter| filter.call instance }
20      instance
21    end
22  end
23end

26RSpec.describe 'css_classes' do
27  let(:klass) { Class.new ApplicationController }
28
29  it 'is a class method of subclasses of ApplicationController' do
30    expect(klass).to respond_to :css_classes
31  end
32
33  it "forwards scopes to the before filter" do
34    expect(klass).to receive(:before_filter_options).with(:only => %w(show))
35    klass.css_classes 'product', only: %w(show)
36    expect(klass.invoke_before_filters.body_class).to eq 'product'
37  end
38
39  it "can be unscoped" do
40    expect(klass).to receive(:before_filter_options).with({})
41    klass.css_classes 'product'
42    expect(klass.invoke_before_filters.body_class).to eq 'product'
43  end
44
45  example "multiple css classes and multiple scopes" do
46    expect(klass).to receive(:before_filter_options).with(:only => %w(new edit index))
47    klass.css_classes 'secondary', 'admin_form', only: %w(new edit index)
48    expect(klass.invoke_before_filters.body_class).to eq 'secondary admin_form'
49  end
50
51  example "lots of classes and scopes" do
52    expect(klass).to receive(:before_filter_options).with(:except => ['index', 'create'])
53    klass.css_classes 'admin', 'administrator', 'category', 'business', 'product', except: ['index', 'create']
54    expect(klass.invoke_before_filters.body_class).to eq 'admin administrator category business product'
55  end
