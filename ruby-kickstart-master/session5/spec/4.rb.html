1class Student
  2  include OperatorGeneratorFromSpace
  3
  4  GRADES = [:freshman, :sophmore, :junior, :senior]
  5
  6  attr_accessor :age , :grade
  7
  8  def initialize(age, grade)
  9    @age, @grade = age, grade
 10  end
 11
 12  def <=>(student)
 13    return age <=> student.age unless age == student.age
 14    GRADES.index(grade) <=> GRADES.index(student.grade)
 15  end
 16end
