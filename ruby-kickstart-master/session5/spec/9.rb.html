  5RSpec.describe 'trees_to_html' do
  6  describe 'the html infrastructure' do
  7    let(:html) { Nokogiri::HTML(trees_to_html '') }
  8    def assert_xpath(path)
  9      matches = html.xpath(path)
 10      expect(matches.size).to eq 1
 11      matches.first
 12    end
 13
 14    specify 'doctype is html' do
 15      expect(html.internal_subset.name).to eq 'html'
 16    end
 17
 18    specify 'the root node is <html>' do
 19      assert_xpath '/html'
 20    end
 21
 22    specify 'inside <html>, is <head>' do
 23      assert_xpath '/html/head'
 24    end
 25
 26    specify 'inside <head>, is a <title> of "Green Thumb Nursery"' do
 27      title = assert_xpath '/html/head/title'
 28      expect(title.text.strip).to eq "Green Thumb Nursery"
 29    end
 30
 31    specify 'inside <html>, is a <body>' do
 32      assert_xpath '/html/body'
 33    end
 34
 35    specify 'inside <body>, is an <h1> of "Catalog"' do
 36      h1 = assert_xpath '/html/body/h1'
 37      expect(h1.text.strip).to eq 'Catalog'
 38    end
 39
 40    specify 'after the catalog, is a <table>' do
 41      table = assert_xpath '/html/body/table'
 42      hopefully_table = html.at_xpath('/html/body/h1').next_element
 43      expect(hopefully_table).to eq table
 44    end
 45
 46    describe 'the first row in the table' do
 47      let(:first_row) { html.at_xpath('/html/body/table/tr') }
 48
 49      it 'exists' do
 50        expect(first_row).to_not eq nil
 51      end
 52
 53      it 'has three <td>s' do
 54        expect(first_row.xpath('./td').size).to eq 3
 55      end
 56
 57      describe 'its <td>s' do
 58        let(:tds) { first_row.xpath './td' }
 59        specify 'the first td\'s text is "Type of tree"' do
 60          expect(tds[0].text.strip).to eq "Type of tree"
 61        end
 62        specify 'the second td\'s text is "Size of the tree"' do
 63          expect(tds[1].text.strip).to eq "Size of the tree"
 64        end
 65        specify 'the third td\'s text is "Price of the tree"' do
 66          expect(tds[2].text.strip).to eq "Price of the tree"
 67        end
 68      end
 69    end
 70  end
